        /*
    Given an integer array 'a' sorted in non-decreasing order, return an array of squares of each number
    sorted in non-decreasing order.
*/

#include <iostream>
#include <vector>
using namespace std;

void squareOfArray(vector &v) {
    for(int i = 0; i < v.size(); i++) {
        v[i] = v[i] * v[i];
    }

    for(int i = 0; i < v.size(); i++) {
        for(int j = i+1; j < v.size(); j++) {
            if(v[i] > v[j]) {
                int temp = v[i];
                v[i] = v[j];
                v[j] = temp;
            }
        }
    }

    // Printing the sorted array

    cout << "The square array is : \n";

    for(int i = 0; i < v.size(); i++) {
        cout << v[i] << " " << endl;
    }
}
 
int main() {

    vector v;
    int n;

    cout << "Enter the size of array : ";
    cin >> n;

    // Taking the input from user

    cout << "Enter the elements in array sorted in ascending order : \n";

    for(int i = 0; i < n; i++) {
        int element;
        cin >> element;
        v.push_back(element);
    }

    squareOfArray(v);

    return 0;
}
